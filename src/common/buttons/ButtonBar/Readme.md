### Simple Button
```jsx
const Button = require('../Button').default;

    <ButtonBar>
        <Button>Click Here</Button>
    </ButtonBar>
```

### Left justified Button
```jsx
const Button = require('../Button').default;
    <ButtonBar justify='left'> 
        <Button>Click Here</Button>
    </ButtonBar>
```

### Right justified Button
```jsx
const Button = require('../Button').default;
    <ButtonBar justify='right'> 
        <Button> Click Here</Button>
    </ButtonBar>
```
